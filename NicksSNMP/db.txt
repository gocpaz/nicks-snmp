CREATE TABLE MIB
(
    MIB_ID INT NOT NULL GENERATED ALWAYS AS IDENTITY CONSTRAINT MIB_PK PRIMARY KEY ,
    MIB_NAME VARCHAR(50) NOT NULL
);


CREATE TABLE OID
(
    OID_ID INT NOT NULL GENERATED ALWAYS AS IDENTITY CONSTRAINT OID_PK PRIMARY KEY ,
    OID_NAME VARCHAR(50) NOT NULL,
    OID_INDEX VARCHAR(50) NOT NULL,
    OWNER_MIB INT NOT NULL,
    CONSTRAINT OWNER_MIB_FK FOREIGN KEY (OWNER_MIB) REFERENCES MIB(MIB_ID)
);

CREATE TABLE IF_TABLE_RFC1213
(
	IF_TABLE_ID INT NOT NULL GENERATED ALWAYS AS IDENTITY CONSTRAINT IF_TABLE_PK PRIMARY KEY ,
	 IF_INDEX INT,
	 IF_IN_OCTETS BIGINT	,
	 IF_IN_UCAST_PKTS	BIGINT,
	 IF_IN_NUCAST_PKTS	BIGINT,
	 IF_IN_DISCARDS	BIGINT,
	 IF_IN_ERRORS BIGINT,
	 IF_IN_UNKNOWN_PROTOS BIGINT,
	 IF_OUT_OCTETS	BIGINT,
	 IF_OUT_UCAST_PKTS	BIGINT,
	 IF_OUT_NUCAST_PKTS BIGINT,	
	 IF_OUT_DISCARDS	BIGINT,
	 IF_DESCR VARCHAR(250)	,
	 IF_OUT_ERRORS BIGINT,	
	 IF_OUT_Q_LEN BIGINT	,
	 IF_SPECIFIC VARCHAR(250),
	 IF_TYPE BIGINT	,
	 IF_MTU BIGINT	,
	 IF_SPEED BIGINT,	
	 IF_PHYS_ADDRESS VARCHAR(250)	,
	 IF_ADMIN_STATUS BIGINT	,
	 IF_OPER_STATUS BIGINT	,
	 IF_LAST_CHANGE VARCHAR(250)	
);

CREATE TABLE TCP_CONN_TABLE_RFC1213
(
	TCP_CONN_TABLE_ID INT NOT NULL GENERATED ALWAYS AS IDENTITY CONSTRAINT TCP_CONN_TABLE_PK PRIMARY KEY ,
	 STATE INT,
	 LOCAL_ADDRESS VARCHAR(250)	,
	 LOCAL_PORT INT,
	 REM_ADDRESS VARCHAR(250)	,
	 REM_PORT INT
 )
